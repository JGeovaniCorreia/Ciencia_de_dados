## Usage

`/serve-model [output_path]`

Default output: `serve.py` in the project root.

## What to generate

Create a self-contained `serve.py` (or the specified path) with the following structure:

```python
"""
FastAPI serving script for the California Housing price prediction pipeline.
Generated by /serve-model.

Run:
    uvicorn serve:app --host 0.0.0.0 --port 8000

Dependencies:
    pip install fastapi uvicorn joblib numpy pandas scikit-learn catboost
"""

from contextlib import asynccontextmanager
from pathlib import Path
import json
import joblib
import numpy as np
import pandas as pd
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, Field

# --- Config ---
PIPELINE_PATH = Path("artifacts/competition_winner_catboost.joblib")
METADATA_PATH = Path("artifacts/competition_metadata.json")

# --- Lifespan: load once at startup ---
_state: dict = {}

@asynccontextmanager
async def lifespan(app: FastAPI):
    _state["pipeline"] = joblib.load(PIPELINE_PATH)
    with open(METADATA_PATH) as f:
        meta = json.load(f)
    _state["q_hats"] = {float(k): v for k, v in meta["conformal_q_hats"].items()}
    _state["model"] = meta["competition_winner"]
    _state["training_date"] = meta["training_date"][:10]
    yield
    _state.clear()

app = FastAPI(title="California Housing Price Predictor", lifespan=lifespan)

# --- Input schema ---
class HousingInput(BaseModel):
    RendaMediana: float = Field(..., ge=0.5, le=15.0, description="Median income ($10k units)")
    IdadeMediaResidencias: float = Field(..., ge=1, le=52)
    MediaComodos: float = Field(..., ge=1.0, le=50.0)
    MediaQuartos: float = Field(..., ge=0.5, le=10.0)
    Populacao: float = Field(..., ge=3, le=35682)
    MediaOcupacao: float = Field(..., ge=0.5, le=20.0)
    Latitude: float = Field(..., ge=32.5, le=42.0)
    Longitude: float = Field(..., le=-114.3, ge=-124.4)
    confidence: float = Field(0.8, ge=0.0, le=1.0, description="Interval confidence: 0.8, 0.9, or 0.95")

# --- Output schema ---
class PredictionOutput(BaseModel):
    prediction_usd: float
    interval_low_usd: float | None
    interval_high_usd: float | None
    confidence: float
    model: str
    training_date: str

# --- Endpoints ---
@app.get("/health")
def health():
    return {
        "status": "ok",
        "model": _state.get("model"),
        "training_date": _state.get("training_date"),
        "pipeline_path": str(PIPELINE_PATH),
    }

@app.post("/predict", response_model=PredictionOutput)
def predict(payload: HousingInput):
    X = pd.DataFrame([payload.model_dump(exclude={"confidence"})])
    try:
        log_pred = _state["pipeline"].predict(X)[0]
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Prediction failed: {e}")

    pred_usd = float(np.expm1(log_pred) * 100_000)

    q_hats = _state["q_hats"]
    available = sorted(q_hats.keys())
    closest = min(available, key=lambda k: abs(k - payload.confidence))
    q_hat = q_hats.get(closest)

    if q_hat is not None:
        low = float(np.expm1(log_pred - q_hat) * 100_000)
        high = float(np.expm1(log_pred + q_hat) * 100_000)
    else:
        low = high = None

    return PredictionOutput(
        prediction_usd=round(pred_usd, 2),
        interval_low_usd=round(low, 2) if low else None,
        interval_high_usd=round(high, 2) if high else None,
        confidence=closest,
        model=_state["model"],
        training_date=_state["training_date"],
    )
```

## After generating the file

Check if `fastapi` and `uvicorn` are installed:
```powershell
.venv\Scripts\pip show fastapi uvicorn 2>&1
```

If not installed:
```powershell
.venv\Scripts\pip install fastapi uvicorn
```

Then print the startup command and a curl example:
```bash
# Start server
uvicorn serve:app --host 0.0.0.0 --port 8000

# Test prediction
curl -X POST http://localhost:8000/predict \
  -H "Content-Type: application/json" \
  -d '{"RendaMediana":5.0,"IdadeMediaResidencias":20.0,"MediaComodos":5.5,"MediaQuartos":1.1,"Populacao":1200.0,"MediaOcupacao":2.8,"Latitude":34.05,"Longitude":-118.24,"confidence":0.8}'
```
